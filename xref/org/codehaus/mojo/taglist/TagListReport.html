200     /**
201      * Defines each tag class (grouping) and the individual tags within each class. The user can also specify a title
202      * for each tag class and the matching logic used by each tag.
203      * <ul>
204      * <li><b>Exact Match</b> <br/>
205      * &lt;matchString&gt;todo&lt;/matchString&gt;<br/>
206      * &lt;matchType&gt;exact&lt;/matchType&gt; <br/>
207      * <i>Matches: todo </i></li>
208      * <li><b>Ignore Case Match</b> <br/>
209      * &lt;matchString&gt;todo&lt;/matchString&gt;<br/>
210      * &lt;matchType&gt;ignoreCase&lt;/matchType&gt; <br/>
211      * <i>Matches: todo, Todo, TODO... </i></li>
212      * <li><b>Regular Expression Match</b> <br/>
213      * &lt;matchString&gt;tod[aeo]&lt;/matchString&gt;<br/>
214      * &lt;matchType&gt;regEx&lt;/matchType&gt; <br/>
215      * <i>Matches: toda, tode, todo </i></li>
216      * </ul>
217      * <br/>
218      * <br/>
219      * For complete examples see the <a href="usage.html"><b>Usage</b></a> page. <br/>
220      * <br/>
221      * The legacy tags configuration remains for backwards compatibility, and the simultaneous use of both the tags and
222      * tagListOptions is permitted; however the tags configuration should be avoided whenever possible because those
223      * strings are only checked using the exact match logic.
224      * 
225      * @parameter
226      * @since 2.4
227      */

230     /**
231      * {@inheritDoc}
232      * 
233      * @see org.apache.maven.reporting.AbstractMavenReport#executeReport(java.util.Locale)
234      */
235     protected void executeReport( Locale locale )
236         throws MavenReportException
237     {
238         this.currentLocale = locale;
239 
240         // User entered no tags and no tagOptions, then default tags
241         if ( ( tags == null || tags.length == 0 )
242             && ( tagListOptions == null || tagListOptions.getTagClasses().size() == 0 ) )
243         {
244             tags = new String[] { "@todo", "TODO" };
245         }
246 
247         if ( StringUtils.isEmpty( encoding ) )
248         {
249             getLog().warn(
250                            "File encoding has not been set, using platform encoding "
251                                + System.getProperty( "file.encoding" ) + ", i.e. build is platform dependent!" );
252         }
253 
254         // Create the tag classes
255         ArrayList tagClasses = new ArrayList();
256 
257         // If any old style tags were used, then add each tag as a tag class
258         if ( tags != null && tags.length > 0 )
259         {
260             getLog().warn( "Using legacy tag format.  This is not recommended." );
261             for ( int i = 0; i < tags.length; i++ )
262             {
263                 TagClass tc = new TagClass( tags[i] );
264                 try
265                 {
266                     AbsTag newTag = TagFactory.createTag( "exact", tags[i] );
267                     tc.addTag( newTag );
268 
269                     tagClasses.add( tc );
270                 }
271                 catch ( InvalidTagException e )
272                 {
273                     // This should be impossible since exact is supported.
274                     getLog().error( "Invalid tag type used.  tag type: exact" );
275                 }
276             }
277         }
278 
279         // If the new style of tag options were used, add them
280         if ( tagListOptions != null && tagListOptions.getTagClasses().size() > 0 )
281         {
282             // Scan each tag class
283             Iterator classIter = tagListOptions.getTagClasses().iterator();
284             while ( classIter.hasNext() )
285             {
286                 org.codehaus.mojo.taglist.options.TagClass tcOption =
287                     (org.codehaus.mojo.taglist.options.TagClass) classIter.next();
288 
289                 // Store the tag class display name.
290                 TagClass tc = new TagClass( tcOption.getDisplayName() );
291 
292                 // Scan each tag within this tag class.
293                 Iterator tagIter = tcOption.getTags().iterator();
294                 while ( tagIter.hasNext() )
295                 {
296                     org.codehaus.mojo.taglist.options.Tag tagOption =
297                         (org.codehaus.mojo.taglist.options.Tag) tagIter.next();
298 
299                     // If a match type is not specified use default.
300                     String matchType = tagOption.getMatchType();
301                     if ( matchType == null || matchType.length() == 0 )
302                     {
303                         matchType = TagFactory.getDefaultTagType();
304                     }
305 
306                     try
307                     {
308                         // Create the tag based on the match type, and add it to the tag class
309                         AbsTag newTag = TagFactory.createTag( matchType, tagOption.getMatchString() );
310                         tc.addTag( newTag );
311                     }
312                     catch ( InvalidTagException e )
313                     {
314                         // This should be impossible since exact is supported.
315                         getLog().error( "Invalid tag type used.  tag type: " + matchType );
316                     }
317                 }
318 
319                 // Add this new tag class to the container.
320                 tagClasses.add( tc );
321             }
322         }
323 
324         // let's proceed to the analysis
325         FileAnalyser fileAnalyser = new FileAnalyser( this, tagClasses );
326         Collection tagReports = fileAnalyser.execute();
327 
328         // Renders the report
329         ReportGenerator generator = new ReportGenerator( this, tagReports );
330         if ( linkXRef )
331         {
332             String relativePath = getRelativePath( xrefLocation );
333             if ( xrefLocation.exists() )
334             {
335                 // XRef was already generated by manual execution of a lifecycle binding
336                 generator.setXrefLocation( relativePath );
337                 generator.setTestXrefLocation( getRelativePath( testXrefLocation ) );
338             }
339             else
340             {
341                 // Not yet generated - check if the report is on its way
342                 for ( Iterator reports = project.getReportPlugins().iterator(); reports.hasNext(); )
343                 {
344                     ReportPlugin report = (ReportPlugin) reports.next();
345 
346                     String artifactId = report.getArtifactId();
347                     if ( "maven-jxr-plugin".equals( artifactId ) || "jxr-maven-plugin".equals( artifactId ) )
348                     {
349                         getLog().error(
350                                         "Taglist plugin MUST be executed after the JXR plugin."
351                                             + "  No links to xref were generated." );
352                     }
353                 }
354             }
355 
356             if ( generator.getXrefLocation() == null )
357             {
358                 getLog().warn( "Unable to locate Source XRef to link to - DISABLED" );
359             }
360         }
361         generator.generateReport();
362 
363         // Generate the XML report
364         generateXmlReport( tagReports );
365     }

367     /**
368      * Generate an XML report that can be used by other plugins like the dashboard plugin.
369      * 
370      * @param tagReports a collection of the tag reports to be output.
371      */
372     private void generateXmlReport( Collection tagReports )
373     {
374         TagListXMLReport report = new TagListXMLReport();
375         report.setModelEncoding( getEncoding() );
376 
377         // Iterate through each tag and populate an XML tag object.
378         for ( Iterator ite = tagReports.iterator(); ite.hasNext(); )
379         {
380             TagReport tagReport = (TagReport) ite.next();
381 
382             TagListXMLTag tag = new TagListXMLTag();
383             tag.setName( tagReport.getTagName() );
384             tag.setCount( Integer.toString( tagReport.getTagCount() ) );
385 
386             // Iterate though each file that contains the current tag and generate an
387             // XML file object within the current XML tag object.
388             for ( Iterator fite = tagReport.getFileReports().iterator(); fite.hasNext(); )
389             {
390                 FileReport fileReport = (FileReport) fite.next();
391 
392                 TagListXMLFile file = new TagListXMLFile();
393                 file.setName( fileReport.getClassName() );
394                 file.setCount( Integer.toString( fileReport.getLineIndexes().size() ) );
395 
396                 // Iterate though each comment that contains the tag and generate an
397                 // XML comment object within the current xml file object.
398                 for ( Iterator cite = fileReport.getLineIndexes().iterator(); cite.hasNext(); )
399                 {
400                     Integer lineNumber = (Integer) cite.next();
401 
402                     TagListXMLComment comment = new TagListXMLComment();
403                     comment.setLineNumber( Integer.toString( lineNumber.intValue() ) );
404                     comment.setComment( fileReport.getComment( lineNumber ) );
405 
406                     file.addComment( comment );
407                 }
408                 tag.addFile( file );
409             }
410             report.addTag( tag );
411         }
412 
413         // Create the writer for the XML output file.
414         xmlOutputDirectory.mkdirs();
415         File xmlFile = new File( xmlOutputDirectory, "taglist.xml" );
416         FileOutputStream fos = null;
417         OutputStreamWriter output = null;
418 
419         try
420         {
421             fos = new FileOutputStream( xmlFile );
422             output = new OutputStreamWriter( fos, getEncoding() );
423 
424             // Write out the XML output file.
425             TaglistOutputXpp3Writer xmlWriter = new TaglistOutputXpp3Writer();
426             xmlWriter.write( output, report );
427         }
428         catch ( Exception e )
429         {
430             getLog().warn( "Could not save taglist xml file: " + e.getMessage() );
431         }
432         finally
433         {
434             IOUtil.close( output );
435         }
436     }

471     /**
472      * Removes empty dirs from the list.
473      * 
474      * @param sourceDirectories the original list of directories.
475      * @return a new list containing only non empty dirs.
476      */
477     private List pruneSourceDirs( List sourceDirectories )
478     {
479         List pruned = new ArrayList( sourceDirectories.size() );
480         for ( Iterator i = sourceDirectories.iterator(); i.hasNext(); )
481         {
482             String dir = (String) i.next();
483             if ( !pruned.contains( dir ) && hasSources( new File( dir ) ) )
484             {
485                 pruned.add( dir );
486             }
487         }
488         return pruned;
489     }

491     /**
492      * Checks whether the given directory contains Java files.
493      * 
494      * @param dir the source directory.
495      * @return true if the folder or one of its subfolders contains at least 1 Java file.
496      */
497     private boolean hasSources( File dir )
498     {
499         boolean found = false;
500         if ( dir.exists() && dir.isDirectory() )
501         {
502             File[] files = dir.listFiles();
503             for ( int i = 0; i < files.length && !found; i++ )
504             {
505                 File currentFile = files[i];
506                 if ( currentFile.isFile() && currentFile.getName().endsWith( ".java" ) )
507                 {
508                     found = true;
509                 }
510                 else if ( currentFile.isDirectory() )
511                 {
512                     boolean hasSources = hasSources( currentFile );
513                     if ( hasSources )
514                     {
515                         found = true;
516                     }
517                 }
518             }
519         }
520         return found;
521     }

523     /**
524      * Construct the list of source directories to analyze.
525      * 
526      * @return the list of dirs.
527      */
528     public List constructSourceDirs()
529     {
530         List dirs = new ArrayList( project.getCompileSourceRoots() );
531         if ( !skipTestSources )
532         {
533             dirs.addAll( project.getTestCompileSourceRoots() );
534         }
535 
536         if ( aggregate )
537         {
538             for ( Iterator i = reactorProjects.iterator(); i.hasNext(); )
539             {
540                 MavenProject reactorProject = (MavenProject) i.next();
541 
542                 if ( "java".equals( reactorProject.getArtifact().getArtifactHandler().getLanguage() ) )
543                 {
544                     dirs.addAll( reactorProject.getCompileSourceRoots() );
545                     if ( !skipTestSources )
546                     {
547                         dirs.addAll( reactorProject.getTestCompileSourceRoots() );
548                     }
549                 }
550             }
551         }
552 
553         dirs = pruneSourceDirs( dirs );
554         return dirs;
555     }
