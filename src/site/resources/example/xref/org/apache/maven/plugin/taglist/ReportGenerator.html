19  import java.util.ArrayList;
20  import java.util.Collection;
21  import java.util.Collections;
22  import java.util.Iterator;
23  import java.util.ResourceBundle;
24  
25  import org.apache.maven.plugin.taglist.beans.FileReport;
26  import org.apache.maven.plugin.taglist.beans.TagReport;
27  import org.apache.maven.reporting.MavenReportException;
28  import org.codehaus.doxia.sink.Sink;

30  /***
31   * Generates the taglist report using Doxia.
32   * 
33   * @author <a href="mailto:bellingard@gmail.com">Fabrice Bellingard </a>
34   */
35  
36  public class ReportGenerator
37  {
38  
39      /***
40       * Generates the whole report using each tag reports made during the
41       * analysis.
42       * 
43       * @param tagReports
44       *            a Collection of TagReport objects
45       * @param locale
46       *            the user locale
47       */
48      public static void generateReport( Collection tagReports, ResourceBundle bundle, Sink sink )
49          throws MavenReportException
50      {
51          sink.head();
52          sink.title();
53          sink.text( bundle.getString( "report.taglist.header" ) );
54          sink.title_();
55          sink.head_();
56  
57          sink.body();
58          sink.section1();
59  
60          sink.sectionTitle1();
61          sink.text( bundle.getString( "report.taglist.mainTitle" ) );
62          sink.sectionTitle1_();
63  
64          // sort the tag reports by name
65          ArrayList sortedTagReports = new ArrayList( tagReports );
66          Collections.sort( sortedTagReports );
67  
68          // Summary section
69          doSummarySection( sortedTagReports, bundle, sink );
70  
71          // Detail section
72          doDetailSection( sortedTagReports, bundle, sink );
73  
74          sink.section1_();
75          sink.body_();
76          sink.flush();
77          sink.close();
78      }
79  
80      /***
81       * @param tagReports
82       * @param sink
83       */
84      private static void doSummarySection( Collection tagReports, ResourceBundle bundle, Sink sink )
85      {
86          sink.paragraph();
87          sink.text( bundle.getString( "report.taglist.summary.description" ) );
88          sink.paragraph_();
89  
90          sink.table();
91          sink.tableRow();
92          sink.tableHeaderCell();
93          sink.text( bundle.getString( "report.taglist.summary.tag" ) );
94          sink.tableHeaderCell_();
95          sink.tableHeaderCell();
96          sink.text( bundle.getString( "report.taglist.summary.occurences" ) );
97          sink.tableHeaderCell_();
98          sink.tableRow_();
99          for ( Iterator iter = tagReports.iterator(); iter.hasNext(); )
100         {
101             doTagSummary( sink, (TagReport) iter.next() );
102         }
103         sink.table_();
104     }
105 
106     /***
107      * @param sink
108      * @param tagReport
109      */
110     private static void doTagSummary( Sink sink, TagReport tagReport )
111     {
112         sink.tableRow();
113         sink.tableCell();
114         sink.link( "#" + tagReport.getTagName() );
115         sink.text( tagReport.getTagName() );
116         sink.link_();
117         sink.tableCell_();
118         sink.tableCell();
119         sink.text( tagReport.getTagCount() + "" );
120         sink.tableCell_();
121         sink.tableRow_();
122     }
123 
124     /***
125      * @param tagReports
126      * @param sink
127      */
128     private static void doDetailSection( Collection tagReports, ResourceBundle bundle, Sink sink )
129     {
130         sink.paragraph();
131         sink.text( bundle.getString( "report.taglist.detail.description" ) );
132         sink.paragraph_();
133 
134         for ( Iterator iter = tagReports.iterator(); iter.hasNext(); )
135         {
136             // TODO Eclipse/IntelliJ-like todos can be placed anywhere in the code
137             doTagDetailedPart( sink, (TagReport) iter.next(), bundle );
138         }
139     }
140 
141     /***
142      * @param sink
143      * @param tagReport
144      */
145     private static void doTagDetailedPart( Sink sink, TagReport tagReport, ResourceBundle bundle )
146     {
147         sink.section2();
148         sink.sectionTitle2();
149         sink.anchor( tagReport.getTagName() );
150         sink.text( tagReport.getTagName() );
151         sink.anchor_();
152         sink.sectionTitle2_();
153         sink.paragraph();
154         sink.bold();
155         sink.text( bundle.getString( "report.taglist.detail.numberOfOccurrences" ) + tagReport.getTagCount() );
156         sink.bold_();
157         sink.paragraph_();
158 
159         // TODO You can have multiple line
160         // todos, that works also.
161         Collection fileReports = tagReport.getFileReports();
162         ArrayList sortedFileReports = new ArrayList( fileReports );
163         Collections.sort( sortedFileReports );
164 
165         for ( Iterator iter = sortedFileReports.iterator(); iter.hasNext(); )
166         {
167             doFileDetailedPart( sink, (FileReport) iter.next(), bundle );
168         }
169 
170         sink.section2_();
171     }
172 
173     /***
174      * @param sink
175      * @param fileReport
176      */
177     private static void doFileDetailedPart( Sink sink, FileReport fileReport, ResourceBundle bundle )
178     {
179         sink.table();
180         sink.tableRow();
181         sink.tableHeaderCell();
182         sink.text( fileReport.getClassName() );
183         sink.tableHeaderCell_();
184         sink.tableHeaderCell();
185         sink.text( bundle.getString( "report.taglist.detail.line" ) );
186         sink.tableHeaderCell_();
187         sink.tableRow_();
188         for ( Iterator iter = fileReport.getLineIndexes().iterator(); iter.hasNext(); )
189         {
190             doCommentLine( sink, fileReport, (Integer) iter.next() );
191         }
192         sink.table_();
193     }
194 
195     /***
196      * @param sink
197      * @param fileReport
198      * @param lineNumber
199      */
200     private static void doCommentLine( Sink sink, FileReport fileReport, Integer lineNumber )
201     {
202         sink.tableRow();
203         sink.tableCell();
204         sink.text( fileReport.getComment( lineNumber ) );
205         sink.tableCell_();
206         sink.tableCell();
207         sink.link( "xref/" + fileReport.getClassNameWithSlash() + ".html#" + lineNumber );
208         sink.text( lineNumber + "" );
209         sink.link_();
210         sink.tableCell_();
211         sink.tableRow_();
212     }
